59 __host__ __device__ inline float3 GetPoint3d(int i, float depth, RgbdSensor sensor) {
60  int v = i / sensor.cols;
61  int u = i - sensor.cols * v;
62  float3 point;
63  point.z = depth;
64  point.x = (static_cast<float>(u) - sensor.cx) * point.z / sensor.fx;
65  point.y = (static_cast<float>(v) - sensor.cy) * point.z / sensor.fy;
